## üåà Color Usage

### Available Colors
```kotlin
// Primary Colors
AppColors.Primary          // Orange #FF6B35 - Main brand color
AppColors.PrimaryDark      // Orange #FF5722 - Darker variant
AppColors.PrimaryLight     // Orange #FFE0B2 - Light variant

// Status Colors
AppColors.Pickup          // Green #4CAF50 - Pickup locations
AppColors.Drop            // Red #F44336 - Drop locations

// Text Colors
AppColors.TextPrimary     // Dark Gray #212121 - Main text
AppColors.TextSecondary   // Medium Gray #757575 - Secondary text
AppColors.TextHint        // Light Gray #BDBDBD - Hints, placeholders

// Background Colors
AppColors.Background      // Light Gray #FAFAFA - Screen background
AppColors.Surface         // White #FFFFFF - Card, surface background

// Borders & Dividers
AppColors.Border          // Gray #E0E0E0 - Input borders
AppColors.Divider         // Gray #EEEEEE - Dividers
```

### Usage Examples
```kotlin
// ‚úÖ CORRECT - Using AppColors
Text(
    text = "Hello",
    color = AppColors.TextPrimary
)

Card(
    colors = CardDefaults.cardColors(
        containerColor = AppColors.Surface
    )
) { }

Button(
    colors = ButtonDefaults.buttonColors(
        containerColor = AppColors.Primary
    )
) { }

// ‚ùå WRONG - Hardcoded colors
Text(
    text = "Hello",
    color = Color(0xFF212121) // Don't do this!
)

Button(
    colors = ButtonDefaults.buttonColors(
        containerColor = Color(0xFFFF6B35) // Don't do this!
    )
) { }
```

## üì± Screen Patterns

### Pattern 1: Form Screen
```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun MyFormScreen(
    onSubmit: () -> Unit,
    onBack: () -> Unit
) {
    var name by remember { mutableStateOf("") }
    var phone by remember { mutableStateOf("") }
    var showError by remember { mutableStateOf(false) }

    val focusManager = LocalFocusManager.current

    Scaffold(
        topBar = {
            TopAppBar(
                title = {
                    Text(
                        text = "Form Title",
                        style = MaterialTheme.typography.titleLarge,
                        fontWeight = FontWeight.Bold
                    )
                },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(
                            imageVector = Icons.Default.ArrowBack,
                            contentDescription = "Back",
                            tint = AppColors.TextPrimary
                        )
                    }
                },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = Color.White
                )
            )
        },
        containerColor = AppColors.Background
    ) { paddingValues ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
        ) {
            // Scrollable Content
            Column(
                modifier = Modifier
                    .weight(1f)
                    .verticalScroll(rememberScrollState())
                    .padding(16.dp)
            ) {
                SectionHeader(text = "Personal Information")

                Spacer(modifier = Modifier.height(16.dp))

                NameInputField(
                    value = name,
                    onValueChange = { name = it },
                    isError = showError && name.isBlank(),
                    errorMessage = "Name is required",
                    keyboardActions = KeyboardActions(
                        onNext = { focusManager.moveFocus(FocusDirection.Down) }
                    )
                )

                Spacer(modifier = Modifier.height(16.dp))

                PhoneNumberField(
                    value = phone,
                    onValueChange = { phone = it },
                    isError = showError && phone.length != 10,
                    errorMessage = "Enter valid phone",
                    keyboardActions = KeyboardActions(
                        onDone = { focusManager.clearFocus() }
                    )
                )
            }

            // Bottom Button
            Surface(
                color = Color.White,
                shadowElevation = 8.dp
            ) {
                PrimaryButton(
                    text = "Submit",
                    onClick = {
                        if (name.isBlank() || phone.length != 10) {
                            showError = true
                        } else {
                            onSubmit()
                        }
                    },
                    icon = Icons.Default.Check,
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(16.dp)
                )
            }
        }
    }
}
```

### Pattern 2: List Screen
```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun MyListScreen(
    items: List<MyItem>,
    isLoading: Boolean,
    onItemClick: (MyItem) -> Unit,
    onBack: () -> Unit
) {
    Scaffold(
        topBar = {
            TopAppBar(
                title = {
                    Text(
                        text = "My List",
                        style = MaterialTheme.typography.titleLarge,
                        fontWeight = FontWeight.Bold
                    )
                },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(
                            imageVector = Icons.Default.ArrowBack,
                            contentDescription = "Back",
                            tint = AppColors.TextPrimary
                        )
                    }
                },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = Color.White
                )
            )
        },
        containerColor = AppColors.Background
    ) { paddingValues ->
        when {
            isLoading -> {
                LoadingIndicator(
                    message = "Loading items...",
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(paddingValues)
                )
            }
            items.isEmpty() -> {
                EmptyState(
                    icon = Icons.Default.Inbox,
                    title = "No items found",
                    subtitle = "Items will appear here",
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(paddingValues)
                        .padding(32.dp)
                )
            }
            else -> {
                LazyColumn(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(paddingValues),
                    contentPadding = PaddingValues(16.dp),
                    verticalArrangement = Arrangement.spacedBy(12.dp)
                ) {
                    items(items) { item ->
                        InfoCard(
                            modifier = Modifier
                                .fillMaxWidth()
                                .clickable { onItemClick(item) }
                        ) {
                            // Item content
                            Text(item.title)
                        }
                    }
                }
            }
        }
    }
}
```

### Pattern 3: Detail Screen
```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun MyDetailScreen(
    item: MyItem?,
    onEdit: () -> Unit,
    onDelete: () -> Unit,
    onBack: () -> Unit
) {
    if (item == null) {
        EmptyState(
            icon = Icons.Default.Warning,
            title = "Item not found",
            actionText = "Go Back",
            onAction = onBack
        )
        return
    }

    Scaffold(
        topBar = {
            TopAppBar(
                title = {
                    Text(
                        text = "Details",
                        style = MaterialTheme.typography.titleLarge,
                        fontWeight = FontWeight.Bold
                    )
                },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(
                            imageVector = Icons.Default.ArrowBack,
                            contentDescription = "Back",
                            tint = AppColors.TextPrimary
                        )
                    }
                },
                actions = {
                    IconButton(onClick = onEdit) {
                        Icon(
                            imageVector = Icons.Default.Edit,
                            contentDescription = "Edit",
                            tint = AppColors.Primary
                        )
                    }
                },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = Color.White
                )
            )
        },
        containerColor = AppColors.Background
    ) { paddingValues ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
                .verticalScroll(rememberScrollState())
        ) {
            // Content Cards
            InfoCard(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp)
            ) {
                SectionHeader(text = "Information")
                Spacer(modifier = Modifier.height(16.dp))
                Text(item.description)
            }

            // Action Buttons
            Row(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp),
                horizontalArrangement = Arrangement.spacedBy(12.dp)
            ) {
                SecondaryButton(
                    text = "Delete",
                    onClick = onDelete,
                    icon = Icons.Default.Delete,
                    modifier = Modifier.weight(1f)
                )
                PrimaryButton(
                    text = "Edit",
                    onClick = onEdit,
                    icon = Icons.Default.Edit,
                    modifier = Modifier.weight(1f)
                )
            }
        }
    }
}
```

## ‚úÖ Best Practices

### 1. Always Use Theme Colors
```kotlin
// ‚úÖ GOOD
Text(color = AppColors.TextPrimary)

// ‚ùå BAD
Text(color = Color(0xFF000000))
```

### 2. Use Material Typography
```kotlin
// ‚úÖ GOOD
Text(style = MaterialTheme.typography.bodyMedium)

// ‚ùå BAD
Text(fontSize = 14.sp)
```

### 3. Consistent Spacing
```kotlin
// Use multiples of 4dp or 8dp
Spacer(modifier = Modifier.height(16.dp))  // ‚úÖ
Spacer(modifier = Modifier.height(15.dp))  // ‚ùå

// Standard paddings
.padding(16.dp)    // Cards, Screens
.padding(8.dp)     // Small spacing
.padding(4.dp)     // Minimal spacing
```

### 4. Reuse Components
```kotlin
// ‚úÖ GOOD - Reuse PrimaryButton
PrimaryButton(
    text = "Submit",
    onClick = { },
    icon = Icons.Default.Check
)

// ‚ùå BAD - Create custom button
Button(
    onClick = { },
    colors = ButtonDefaults.buttonColors(
        containerColor = Color(0xFFFF6B35)
    ),
    shape = RoundedCornerShape(16.dp)
) {
    Icon(Icons.Default.Check, null)
    Text("Submit")
}
```

### 5. Consistent Corner Radius
```kotlin
// Standard radius
RoundedCornerShape(16.dp)  // Cards
RoundedCornerShape(12.dp)  // Buttons, Inputs
RoundedCornerShape(8.dp)   // Small elements
```

### 6. Error Handling
```kotlin
// Always handle states
when {
    isLoading -> LoadingIndicator()
    error != null -> ErrorMessageCard(error)
    items.isEmpty() -> EmptyState()
    else -> ContentList()
}
```

## ‚ùå Common Mistakes

### Mistake 1: Hardcoding Colors
```kotlin
// ‚ùå WRONG
Card(colors = CardDefaults.cardColors(
    containerColor = Color(0xFFFFFFFF)
))

// ‚úÖ CORRECT
Card(colors = CardDefaults.cardColors(
    containerColor = AppColors.Surface
))
```

### Mistake 2: Inconsistent Spacing
```kotlin
// ‚ùå WRONG
Spacer(modifier = Modifier.height(13.dp))
Spacer(modifier = Modifier.height(7.dp))

// ‚úÖ CORRECT
Spacer(modifier = Modifier.height(16.dp))
Spacer(modifier = Modifier.height(8.dp))
```

### Mistake 3: Not Using Components
```kotlin
// ‚ùå WRONG
Button(
    onClick = { },
    modifier = Modifier.fillMaxWidth().height(56.dp),
    colors = ButtonDefaults.buttonColors(
        containerColor = AppColors.Primary
    ),
    shape = RoundedCornerShape(16.dp)
) {
    Text("Submit")
}

// ‚úÖ CORRECT
PrimaryButton(
    text = "Submit",
    onClick = { },
    modifier = Modifier.fillMaxWidth()
)
```

### Mistake 4: Forgetting Error States
```kotlin
// ‚ùå WRONG
AppTextField(
    value = text,
    onValueChange = { text = it },
    label = "Name"
)

// ‚úÖ CORRECT
AppTextField(
    value = text,
    onValueChange = { text = it },
    label = "Name",
    isError = text.isBlank() && showError,
    errorMessage = "Name is required"
)
```

### Mistake 5: Ignoring Loading States
```kotlin
// ‚ùå WRONG
Button(onClick = { submit() }) {
    Text("Submit")
}

// ‚úÖ CORRECT
PrimaryButton(
    text = "Submit",
    onClick = { submit() },
    isLoading = isSubmitting,
    enabled = !isSubmitting
)
```
